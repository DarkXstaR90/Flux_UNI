{
  "rules": {
    ".read": false,
    ".write": false,
    "movies": {
      ".read": "auth != null",
      ".write": "auth != null && root.child('users').child(auth.uid).child('role').val() == 'admin'",
      "$movieId": {
        ".validate": "newData.hasChildren(['title','description','thumbnail','vimeoUrl','date','createdAt']) && newData.child('title').isString() && newData.child('title').val().length > 0 && newData.child('description').isString() && newData.child('thumbnail').isString() && newData.child('vimeoUrl').isString() && newData.child('date').isString()"
      }
    },
    "users": {
      "$uid": {
        ".read": "auth != null && auth.uid == $uid",
        ".write": "auth != null && auth.uid == $uid",
        "role": {
          ".write": "false",
          ".validate": "newData.val() == 'user' || newData.val() == 'admin'"
        },
        "username": { ".validate": "newData.isString() && newData.val().length > 0 && newData.val().length <= 40" },
        "likes": { "$movie": { ".validate": "newData.val() === true || !newData.exists()" } },
        "watchlist": { "$movie": { ".validate": "newData.val() === true || !newData.exists()" } }
      }
    }
  }
